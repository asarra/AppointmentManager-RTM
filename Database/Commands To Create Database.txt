CREATE TABLE KONTAKT(
   KontaktID INT PRIMARY KEY     NOT NULL,
   Name           TEXT    NOT NULL,
   PLZ            INT     NOT NULL,
   Strasse        CHAR(50) NOT NULL,
   Ort        CHAR(50) NOT NULL,
   Bemerkung         TEXT,
   Email TEXT NOT NULL,
   Kontaktart INT NOT NULL,
   Telefon INT NOT NULL,
   Firma CHAR(50) NOT NULL
);



CREATE TABLE TERMIN(
   TerminID INT PRIMARY KEY     NOT NULL,
   KontaktID INT NOT NULL,
   Wiederholung INT NOT NULL,
   Thema TEXT NOT NULL,
   Bemerkung TEXT,
   Startzeit TIMESTAMP NOT NULL,
   Endzeit TIMESTAMP,
   Ort CHAR(50) NOT NULL,
   FOREIGN KEY(KontaktID) REFERENCES KONTAKT(KontaktID)
);



CREATE TABLE ERINNERUNG(
   ErinnerID INT PRIMARY KEY NOT NULL,
   TerminID INT NOT NULL,
   Zeitpunkt TIMESTAMP NOT NULL,
   FOREIGN KEY(TerminID) REFERENCES TERMIN(TerminID)
);






































CREATE TABLE CONTACT(
   ContactID INT PRIMARY KEY     NOT NULL,
   Name           TEXT    NOT NULL,
   PostalCode            INT     NOT NULL,
   Street        CHAR(50) NOT NULL,
   City        CHAR(50) NOT NULL,
   Note         TEXT,
   Email TEXT NOT NULL,
   ContactType INT NOT NULL,
   Telephone INT NOT NULL,
   Company CHAR(50) NOT NULL
);



CREATE TABLE APPOINTMENT(
   AppointmentID INT PRIMARY KEY     NOT NULL,
   ContactID INT NOT NULL,
   Repetition INT NOT NULL,
   Topic TEXT NOT NULL,
   Note TEXT,
   StartTime TIMESTAMP NOT NULL,
   EndTime TIMESTAMP,
   City CHAR(50) NOT NULL,
   FOREIGN KEY(ContactID) REFERENCES CONTACT(ContactID)
);



CREATE TABLE REMINDER(
   ReminderID INT PRIMARY KEY NOT NULL,
   AppointmentID INT NOT NULL,
   Time TIMESTAMP NOT NULL,
   FOREIGN KEY(AppointmentID) REFERENCES APPOINTMENT(AppointmentID)
);
